cmake_minimum_required(VERSION 2.8.3)

project(teleop)

find_package(catkin REQUIRED COMPONENTS
  geometry_msgs
  roscpp
)

catkin_package()

include_directories(
  ${catkin_INCLUDE_DIRS}
)

add_executable(teleop_key src/teleop_key.cpp)

target_link_libraries(teleop_key
  ${catkin_LIBRARIES}
)
